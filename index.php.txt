<script>
let startTime = Date.now(), lastLogs = [], errorCount = 0;
let currentPage = 1, totalPages = 1, allHypotheses = [];

setInterval(()=>{
  let e = Math.floor((Date.now()-startTime)/1000);
  document.getElementById('timer').textContent = Math.floor(e/60).toString().padStart(2,'0')+':'+(e%60).toString().padStart(2,'0');
}, 1000);

function updateBugReport() {
  document.getElementById('bug-report').value = `=== GENESIS-ULTRA BUG ===\nSession: <?=$SESSION_ID?>\nTime: ${new Date().toISOString()}\nStatus: ${document.getElementById('status-text').textContent}\nErrors: ${errorCount}\nLogs:\n${lastLogs.slice(-10).map(l=>`  - ${l.msg}`).join('\n')}`;
}
function copyBugReport() {
  document.getElementById('bug-report').select();
  document.execCommand('copy');
  document.getElementById('copy-status').textContent = '✅ Copié!';
  setTimeout(()=>document.getElementById('copy-status').textContent='', 2000);
}

// Charger les découvertes depuis le serveur
function loadHypotheses(page = 1) {
  fetch('agent.php?action=load_hypotheses&page='+page)
  .then(r=>r.json())
  .then(data=>{
    allHypotheses = data.hypotheses || [];
    totalPages = data.total_pages || 1;
    currentPage = page;
    document.getElementById('total-hypo').textContent = data.total_count || 0;
    renderHypotheses(allHypotheses);
    renderPagination(page, totalPages);
  });
}

function renderHypotheses(hypos) {
  if(!hypos || hypos.length === 0) {
    document.getElementById('results').innerHTML = '<em>Aucune découverte archivée pour le moment...</em>';
    return;
  }
  document.getElementById('results').innerHTML = hypos.map((h,i)=>`
    <div class="hypo" onclick="showDetail(${i})">
      <h3>💡 ${h.title || 'Hypothèse'}</h3>
      <p><strong>Cible:</strong> ${h.target || 'N/A'}</p>
      <p><strong>Sources:</strong> ${(h.sources||[]).map(s=>`<span class="src-tag src-${s.toLowerCase()}">${s}</span>`).join(' ')}</p>
      <p><strong>Score:</strong> <span class="badge ${h.novelty>0.7?'badge-high':(h.novelty>0.4?'badge-med':'badge-low')}">${Math.round((h.novelty||0)*100)}%</span></p>
      <p><small>📅 ${h.timestamp ? new Date(h.timestamp*1000).toLocaleString('fr-FR') : 'N/A'}</small></p>
    </div>
  `).join('');
}

function renderPagination(current, total) {
  if(total <= 1) {
    document.getElementById('pagination').innerHTML = '';
    return;
  }
  let html = '';
  for(let i=1; i<=total; i++) {
    html += `<button class="${i===current?'active':''}" onclick="loadHypotheses(${i})">${i}</button>`;
  }
  document.getElementById('pagination').innerHTML = html;
}

function showDetail(index) {
  let h = allHypotheses[index];
  let detail = `=== DÉCOUVERTE #${h.id || index+1} ===\n\n`+
    `🎯 CIBLE: ${h.target}\n\n`+
    `💡 HYPOTHÈSE:\n${h.title}\n\n`+
    `📝 RÉSUMÉ:\n${h.summary || h.hypothesis || 'N/A'}\n\n`+
    `🔗 SOURCES CROISÉES:\n${(h.sources||[]).join(' → ')}\n\n`+
    `🎯 SCORE NOUVEAUTÉ: ${Math.round((h.novelty||0)*100)}%\n\n`+
    `⚙️ ACTIONNABLE:\n${h.actionable || 'À vérifier en laboratoire'}\n\n`+
    `📅 DATE: ${h.timestamp ? new Date(h.timestamp*1000).toLocaleString('fr-FR') : 'N/A'}\n\n`+
    `🔍 STATUT: ${h.status || 'EN_ATTENTE_VERIFICATION'}`;
  alert(detail);
}

function poll() {
  fetch('agent.php?action=observe&session=<?=$SESSION_ID?>')
  .then(r=>r.text())
  .then(text=>{
    if(!text.trim().startsWith('{')) throw new Error('Pas JSON: '+text.substring(0,100));
    return JSON.parse(text);
  })
  .then(data=>{
    errorCount = 0;
    if(data.logs && data.logs.length > 0) {
      lastLogs = data.logs;
      updateBugReport();
      document.getElementById('logs').innerHTML = data.logs.map(l=>{
        let cls = l.type || 'info';
        let detail = l.detail ? `<div class="detail-box">${l.detail}</div>` : '';
        return `<div class="log ${cls}"><small>[${l.time||'--:--'}]</small> ${l.msg}${detail}</div>`;
      }).join('');
      document.getElementById('logs').scrollTop = document.getElementById('logs').scrollHeight;
    }
    if(data.new_hypothesis) {
      loadHypotheses(1); // Recharger la page 1
    }
    if(data.status==='error') {
      document.getElementById('status-text').textContent = '❌ Erreur';
      document.getElementById('status-text').style.color = '#900';
    }
  })
  .catch(err=>{
    errorCount++;
    document.getElementById('logs').innerHTML += `<div class="log error">${err.message}</div>`;
    updateBugReport();
  });
}

window.onload = function() {
  fetch('agent.php?action=init&session=<?=$SESSION_ID?>')
  .then(r=>r.text())
  .then(text=>{
    if(!text.trim().startsWith('{')) throw new Error('Init pas JSON');
    return JSON.parse(text);
  })
  .then(data=>{
    if(data.ok) {
      document.getElementById('status-text').textContent = '🔄 Recherche autonome en cours...';
      document.getElementById('status-text').style.color = '#090';
      loadHypotheses(1); // Charger les découvertes
      setInterval(poll, 3000); // Polling toutes les 3s
    } else throw new Error('Init échouée');
  })
  .catch(err=>{
    document.getElementById('status-text').textContent = '❌ '+err.message;
    document.getElementById('logs').innerHTML = `<div class="log error">${err.message}</div>`;
  });
};
